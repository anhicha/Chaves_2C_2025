125static void ControlarLed(void *pvParameter)
126{
127 while (true)
128 {
129 if (!activar_medicion)

132 }
133 else
134 {
135 if (distancia_actual < 10)
136 { // si la distancia es menor a 10 cm
137 LedsOffAll();

139 else if (distancia_actual >= 10 && distancia_actual < 20)
140 { // si la distancia esta entre 10 y 20 cm
141 LedOn(LED_1);
142 LedOff(LED_2);
143 LedOff(LED_3);
144 }
145 else if (distancia_actual >= 20 && distancia_actual < 30)
146 { // si la distancia esta entre 20 y 30 cm
147 LedOn(LED_2);
148 LedOn(LED_1);
149 LedOff(LED_3);
150 }
151 else if (distancia_actual > 30)
152 { // si la distancia es mayor a 30 cm
153 LedOn(LED_1);
154 LedOn(LED_2);
155 LedOn(LED_3);
156 }
157 }

200 .baud_rate = 9600, //define la velocidad de comunicacion
201 .func_p = FuncUart, //interrupcion isr para el uart,
202 .param_p = NULL};
203 UartInit(&my_uart);
204
205 // config timer
206 timer_config_t timer_medicion = {
207 .timer = TIMER_A,
208 .period = MEDICION_PERIOD_US, // 1s
209 .func_p = FuncTimer,
210 .param_p = NULL};
211 TimerInit(&timer_medicion);
212
213 // crear tareas

217 xTaskCreate(&Display, "Display", 512, NULL, 5, &Display_task_handle);
218
219 // iniciar timer
220 TimerStart(timer_medicion.timer);
221}
