121void app_main(void)
122{
123 /* === Inicialización del ADC === */
124 analog_input_config_t adc_config = {
125 .input = CH1, // Canal 1 de entrada
126 .mode = ADC_SINGLE, // Modo de lectura simple, una conversión cada vez que lo leemos
127 .func_p = NULL, // No se usa callback
128 .param_p = NULL, // No se usa callback
129 };
130 AnalogInputInit(&adc_config); // inicializa el ADC
131
132 /* === Inicialización del DAC === */
133 AnalogOutputInit();
134
135 /* === Inicialización de UART === */
136 serial_config_t uart_config = {
137 .port = UART_PC,
138 .baud_rate = 115200, // velocidad alta para no perder muestras
139 .func_p = UART_NO_INT, // no se usa interrupcion
140 .param_p = NULL,
141 };
142 UartInit(&uart_config); // inicializa la UART
143
144 /* === Inicialización del Timer === */
145 timer_config_t timer_cfg = {
146 .timer = TIMER_A,
147 .period = TIMER_PERIOD_US, // cada dos ms
148 .func_p = TimerISR, // despierta la tarea adc task
149 .param_p = NULL};
150 TimerInit(&timer_cfg); // inicializa el timer
151
152 /* === Crear tarea de muestreo y envío === */
153 xTaskCreate(&ADC_task, "ADC_Task", 2048, NULL, 5, &ADC_task_handle);
154 xTaskCreate(&DAC_task, "DAC_Task", 2048, NULL, 5, &DAC_task_handle);
155
156 TimerStart(timer_cfg.timer);
157}
