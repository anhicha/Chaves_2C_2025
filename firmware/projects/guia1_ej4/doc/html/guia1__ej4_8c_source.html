68int8_t convertToBcdArray(uint32_t data, uint8_t digits, uint8_t *bcd_number) {
69 // Rellenar todo con ceros primero
70 for (uint8_t i = 0; i < digits; i++) {
71 bcd_number[i] = 0;
72 }
73
74 // Extraer dígitos desde el final
75 for (int8_t i = digits - 1; i >= 0; i--) {
76 bcd_number[i] = data % 10;
77 data /= 10;
78 }
79

81}
